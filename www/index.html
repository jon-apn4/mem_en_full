    function getSortedList(list) {

    function cmp(a, b) {
    if (autoSortKey === "date-desc") return new Date(b.updatedAt) - new Date(a.updatedAt);
    if (autoSortKey === "date-asc") return new Date(a.updatedAt) - new Date(b.updatedAt);
    if (autoSortKey === "title-desc") return b.title.localeCompare(a.title, "ja");
    if (autoSortKey === "title-asc") return a.title.localeCompare(b.title, "ja");
    return 0; // manual: 元の順を保持
    }
    const sortedPinned = autoSortKey !== "manual" ? [...pinned].sort(cmp) : pinned;
    const sortedUnpinned = autoSortKey !== "manual" ? [...unpinned].sort(cmp) : unpinned;
    return [...sortedPinned, ...sortedUnpinned];
    }

    }

    function readFile(file, enc, cb) {

    function parseCSVLine(line) {
    const result=[]; let cur="",inQ=false;
    for(let i=0;i<line.length;i++){
    const c=line[i];
    if(inQ){ if(c==='"'&&line[i+1]==='"'){cur+='"';i++;} else if(c==='"'){inQ=false;} else cur+=c; }
    else { if(c==='"'){inQ=true;} else if(c===','){result.push(cur);cur="";} else cur+=c; }
    }
    result.push(cur); return result;
    }
